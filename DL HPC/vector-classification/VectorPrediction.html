def my_train(clf, epochs, optimizer, criterion, train_data, test_data):

    train_losses = []
    test_losses = []

    # 
    for epoch_step in range(epochs):   
        #     
        epoch_loss = 0.0
        for i, batch in enumerate(train_data):
            X, y = batch
            optimizer.zero_grad()
            prediction = clf(X.to(device))
            loss = criterion(prediction, y.to(device).float())
            epoch_loss += loss.item()
            loss.backward()
            optimizer.step()
            
        #       
        epoch_test_loss = criterion(clf(test_data[0].to(device)), test_data[1].to(device).float()).item()

        train_losses.append(epoch_loss / (i + 1)), test_losses.append(epoch_test_loss)

        sys.stdout.flush()
        sys.stdout.write(f"\rEpoch: {epoch_step + 1}, epoch steps: {i + 1}. Epoch loss train: {np.round(epoch_loss / (i + 1), decimals=2)}; Epoch loss test: {np.round(epoch_test_loss, decimals=2)}")
    return train_losses, test_losses 

def my_train2(clf, epochs, optimizer, criterion, train_data, test_data):

        train_losses.append(epoch_loss / (i + 1)), test_losses.append(epoch_test_loss)

        sys.stdout.flush()

    return train_losses, test_losses

class CustomDataset(torch.utils.data.Dataset):
    def __init__(self, tensors: torch.Tensor):
        super(CustomDataset, self).__init__()
        self.tensors = tensors

    def __getitem__(self, index):   
        return self.tensors[0][index].to(device), self.tensors[1][index].to(device)

    def __len__(self):
        return len(self.tensors[0])

class PrimitiveMLP(nn.Module):
    def __init__(self, image_size):
        super().__init__()
        self.net = nn.Sequential(
            nn.Flatten(),
            nn.Linear(4, 12),
            nn.ReLU(),
            nn.Linear(12, 24),
            nn.ReLU(),
            nn.Linear(24, 2),
            nn.Softmax(dim=1)
        )
    def forward(self, X):
        return self.net(X)
    def name(self):
        return 'Simple_Linear'

class Model(torch.nn.Module):
    def __init__(self, size, out_size, n_params=15):
        super(Model, self).__init__()
        self.layer1 = torch.nn.Linear(size, n_params)
        self.layer2 = torch.nn.Linear(n_params, out_size)
        self.func = torch.nn.ReLU()
        self.act = torch.nn.Sigmoid()

    def forward(self, x):
        return self.act(self.layer2(self.func(self.layer1(x))))
